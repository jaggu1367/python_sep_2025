To utilize the reusable code, we will define methods 

# define a method to compare two numbers
def compare(a,b):
    #compare a and b
    if a>b:
        print(f"{a} is greater than {b}")
    elif a<b:
        print(f"{a} is less than {b}")
    else:
        print(f"{a} is equal to {b}")
		
# defining the variables 
a=1000
b=2000

# calling the method 
compare(a,b)

# call the method without defining variables(directly passing values)
compare(200,300)
